i2c_pca9685: PCA9685 16 Channel 12 bit PWM Output with I2C interface

Source Code: lblx_i2c_pca9685.erl

Description: PCA9685 16 Channel 12 bit PWM Output with I2C interface


groups() -> [output, i2c_device].

version() -> "0.1.0".

Configs:
{i2c_device, {"i2c-1"}},
{i2c_addr, {16#40}},
{out_enbl_pin, {null}}, % Output GPIO pin number for OE signal

{num_of_channels, {1}}, % Number of channels used. Range 1 to 16

{ext_clk, {false}}, % If true, use external clock. If false, use internal clock

{enbl_all_call, {true}}, % If true respond to all-call address
{all_call_addr, {16#E0}}, % All call address, LSB is insignificant

{enbl_sub1_addr, {false}}, % If true respond to sub-address 1
{sub1_addr, {16#E2}}, % Sub address 1, LSB is insignificant

{enbl_sub2_addr, {false}}, % If true respond to sub-address 2
{sub2_addr, {16#E4}}, % Sub address 2, LSB is insignificant

{enbl_sub3_addr, {false}}, % If true respond to sub-address 3
{sub3_addr, {16#E6}}, % Sub address 3, LSB is insignificant

{prescale, {16#1E}}, % Sets the PWM Frequency. Default 200 Hz (0x1E)
% The max freq is 1526 Hz when prescale is set to 0x03
% The min freq is 24 Hz when prescale is set to 0xFF
{invert_out, {true}}, % If true, output driver is active if desired output state is off
% If false, output driver is active if desired output state is on
{ext_driver, {false}}, % Set to true if external driver is used

{out_enbl0_state, {false}}, % Configures all ouptut states when OE pin on chip is high
{out_enbl1_state, {false}} % Configures all output states when OE pin on chip is high

Inputs:
{sleep, {empty, {empty}}}, % If true, device is in low power mode, oscillator is off.
{out_enbl_val, {empty, {empty}}}, % Output Enable pin value
{all_on_off, {empty, {empty}}}, % Turn all channels full On (true) or full Off (false)
% Set to null for no effect
{all_on_count, {empty, {empty}}}, % Clock count from start of phase to turrning on output (0-4095)
{all_off_count, {empty, {empty}}}, % Clock count from turning ouptut on until output is turned off (0-4095)

{on_off, [{empty, {empty}}]}, % Turn channel full On (true) or full Off (false)
% Set to null for no effect
{on_count, [{empty, {empty}}]}, % Clock count from start of phase to turrning on output (0-4095)
{off_count, [{empty, {empty}}]} % Clock count from turning ouptut on until output is turned off (0-4095)

Outputs:
{pwm_freq, {null, []}} % Current PWM frequency. Set via the prescale config value
